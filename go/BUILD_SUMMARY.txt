                          USAGE EXAMPLE

package main

import (
    "context"
    "log"
    "github.com/moemollaei-org/kairos-connect-sdks/go"
)

func main() {
    // Create client
    client, err := kairos.New("kairos_sk_...")
    if err != nil {
        log.Fatal(err)
    }

    ctx := context.Background()

    // List tasks
    tasks, pagination, err := client.Tasks.List(ctx, nil)
    if err != nil {
        log.Fatal(err)
    }

    // Create task
    task, err := client.Tasks.Create(ctx, kairos.CreateTaskInput{
        Title: "New Task",
        Priority: stringPtr("high"),
    })
    if err != nil {
        log.Fatal(err)
    }
}

func stringPtr(s string) *string { return &s }

================================================================================
                        CONFIGURATION OPTIONS
================================================================================

WithBaseURL(url)          - Custom API base URL
WithHTTPClient(client)    - Custom HTTP client
WithMaxRetries(n)         - Retry count (default 3)
WithTimeout(duration)     - Request timeout

Example:
    client, _ := kairos.New(apiKey,
        kairos.WithBaseURL("https://custom.url/v1"),
        kairos.WithMaxRetries(5),
        kairos.WithTimeout(60*time.Second),
    )
